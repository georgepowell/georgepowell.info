var map = function(list, processor) {
	var rtn = [];
	for (var i = 0; i < list.length; i++) {
		rtn[i] = processor(list[i]);
	}
	return rtn;
}


// simple map method, maps a list from one form to another using a 'processor' function
var where = function(list, predicate) {
	var rtn = [];
	for (var i = 0; i < list.length; i++) {
		if(predicate(list[i]))
		    rtn.push(list[i]);
	}
	return rtn;
}


console.log(map([4, 5, 6, 2, 3, 4, 4, 2], Z("a => a * a * a")));
console.log(map([1, 2, 3, 4, 5, 6, 7, 8, 9], Z("a => a % 2 === 0")));
