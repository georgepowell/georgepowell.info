var Z = function (lambda) {
	var parts = lambda.split(" => ");
	var inputs = parts[0].split(" ");
	var expression = parts[1];

	return function () {
		var argArray = [];
		for (var i = 0; i < inputs.length; i++) {
			argArray[inputs[i]] = arguments[i];
			expression = expression.replace(inputs[i], "argArray['" + inputs[i] + "']");
		}
		return eval(expression);
	}
}
